    Sandbox.JS

    JS Sandboxing via iframe sandboxing.

    Installation

    pnpm install sboxjs

    Usage

    <script type="module">
  import { init } from 'sboxjs';
  (async()=>{
    const sbox = await init();
    const value = await sbox.evaluate('return 1'); // {success:true,result:1};
    if (value.error) throw value.error;
    alert(value.result);
  })();
</script>
